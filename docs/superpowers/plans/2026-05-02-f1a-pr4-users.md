> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `/admin/users` UI: list admins, invite a new admin (creates `User` + sends magic link), revoke an admin (transactional `revokedAt` + session purge). Auto-revoke is blocked. Audit `USER_INVITED` and `USER_REVOKED`.

**Architecture:** `features/users/` holds schemas, queries, and actions. Route lives in `app/admin/users/`. UI uses shadcn `<Dialog>` (invite) + `<AlertDialog>` (revoke confirm) + `<Table>`. List ordering: active users first (revoked at the bottom), then by `createdAt DESC`.

**Tech Stack:** Server Components (queries) · Server Actions (mutations) · `react-hook-form` + `zodResolver` · shadcn `<Dialog>`, `<AlertDialog>`, `<Table>`, `<Badge>` · Better Auth `auth.api.signInMagicLink` for invite flow.

import { format, formatDistanceToNow } from "date-fns";
import { ptBR } from "date-fns/locale";

export function formatDate(date: Date | null | undefined): string {
  if (!date) return "";
  return format(date, "dd/MM/yyyy", { locale: ptBR });
}

export function formatDateTime(date: Date | null | undefined): string {
  if (!date) return "";
  return format(date, "dd/MM/yyyy HH:mm", { locale: ptBR });
}

export function formatRelative(date: Date | null | undefined): string {
  if (!date) return "";
  return formatDistanceToNow(date, { addSuffix: true, locale: ptBR });
}

// src/features/users/queries.ts
import "server-only";

import { db } from "@/lib/db";
import type { UserListItem } from "./types";

export async function listUsers(): Promise<UserListItem[]> {
  const users = await db.user.findMany({
    where: { role: "ADMIN" },
    select: {
      id: true,
      email: true,
      name: true,
      createdAt: true,
      revokedAt: true,
      sessions: {
        select: { createdAt: true },
        orderBy: { createdAt: "desc" },
        take: 1,
      },
    },
    orderBy: [{ revokedAt: "asc" }, { createdAt: "desc" }],
  });

  return users.map((u) => ({
    id: u.id,
    email: u.email,
    name: u.name,
    createdAt: u.createdAt,
    revokedAt: u.revokedAt,
    lastAccessAt: u.sessions[0]?.createdAt ?? null,
  }));
}
```

> Note: Postgres orders nulls last for `ASC` by default, which means `revokedAt: null` (active) come first. If your Postgres version surprises you, switch to a CASE expression via `Prisma.sql`.

// src/features/users/actions.ts
"use server";

import { headers } from "next/headers";
import { revalidatePath } from "next/cache";

import { auditLog } from "@/lib/audit/log";
import { auth } from "@/lib/auth/auth";
import { requireAdmin } from "@/lib/auth/helpers";
import { db } from "@/lib/db";
import {
  type InviteUserInput,
  type RevokeUserInput,
  inviteUserSchema,
  revokeUserSchema,
} from "./schemas";

type ActionResult = { success: true } | { success: false; error: string };

function buildRequest(): Request {
  // Server Actions don't expose Request directly; we synthesise one from
  // headers() so audit log captures IP/UA without leaking it elsewhere.
  return new Request("https://internal/admin", { headers: undefined });
}

export async function inviteUserAction(
  input: InviteUserInput,
): Promise<ActionResult> {
  const session = await requireAdmin();

  const parsed = inviteUserSchema.safeParse(input);
  if (!parsed.success) {
    return { success: false, error: "Dados inválidos." };
  }

  const { email, name } = parsed.data;

  const existing = await db.user.findUnique({
    where: { email },
    select: { id: true },
  });
  if (existing) {
    return {
      success: false,
      error: "Já existe um administrador com este e-mail.",
    };
  }

  const created = await db.user.create({
    data: { email, name: name ?? null, emailVerified: true, role: "ADMIN" },
    select: { id: true, email: true, name: true },
  });

  await auditLog.write({
    action: "USER_INVITED",
    actorId: session.user.id,
    actorEmail: session.user.email,
    resourceType: "User",
    resourceId: created.id,
    metadata: { email: created.email, name: created.name },
  });

  // Fire-and-forget magic link. Failure here is non-fatal — the admin can
  // log in via /admin/login by themselves. We catch and audit, but don't roll
  // back the User creation.
  try {
    await auth.api.signInMagicLink({
      body: { email, callbackURL: "/admin" },
      headers: await headers(),
    });
  } catch (error) {
    const msg = error instanceof Error ? error.message : String(error);
    console.error(`[users.invite] magic link dispatch failed: ${msg}`);
  }

  revalidatePath("/admin/users");
  return { success: true };
}

export async function revokeUserAction(
  input: RevokeUserInput,
): Promise<ActionResult> {
  const session = await requireAdmin();

  const parsed = revokeUserSchema.safeParse(input);
  if (!parsed.success) {
    return { success: false, error: "Dados inválidos." };
  }

  if (parsed.data.userId === session.user.id) {
    return { success: false, error: "Você não pode revogar seu próprio acesso." };
  }

  const target = await db.user.findUnique({
    where: { id: parsed.data.userId },
    select: { id: true, email: true, revokedAt: true },
  });

  if (!target) {
    return { success: false, error: "Usuário não encontrado." };
  }

  if (target.revokedAt) {
    return { success: false, error: "Usuário já está revogado." };
  }

  await db.$transaction([
    db.user.update({
      where: { id: target.id },
      data: { revokedAt: new Date() },
    }),
    db.session.deleteMany({ where: { userId: target.id } }),
  ]);

  await auditLog.write({
    action: "USER_REVOKED",
    actorId: session.user.id,
    actorEmail: session.user.email,
    resourceType: "User",
    resourceId: target.id,
    metadata: { email: target.email },
  });

  revalidatePath("/admin/users");
  return { success: true };
}
```

> ⚠️ The `auth.api.signInMagicLink` call uses Better Auth's server-side API. If the function name differs in the installed version, check the Better Auth docs (or run `grep -r "signInMagicLink\|sendMagicLink" node_modules/better-auth/dist`). The semantically equivalent operation is "send the magic link email by email address".

  inviteUserSchema,
} from "@/features/users/schemas";
import { useMessages } from "@/stores/use-content-store";

type Props = {
  onSuccess: () => void;
};

export function InviteUserForm({ onSuccess }: Props) {
  const { admin, common } = useMessages();
  const [isPending, startTransition] = useTransition();

  const form = useForm<InviteUserInput>({
    resolver: zodResolver(inviteUserSchema),
    defaultValues: { email: "", name: "" },
  });

  function onSubmit(values: InviteUserInput) {
    startTransition(async () => {
      const result = await inviteUserAction(values);
      if (result.success) {
        toast.success(admin.users.inviteDialog.success);
        form.reset();
        onSuccess();
      } else {
        toast.error(result.error);
      }
    });
  }

  return (
    <Form {...form}>
      <form
        onSubmit={form.handleSubmit(onSubmit)}
        className="space-y-4"
        data-testid="invite-user-form"
      >
        <FormField
          control={form.control}
          name="email"
          render={({ field }) => (
            <FormItem>
              <FormLabel>{admin.users.inviteDialog.emailLabel}</FormLabel>
              <FormControl>
                <Input
                  type="email"
                  autoComplete="email"
                  placeholder={common.forms.placeholders.email}
                  data-testid="invite-user-email"
                  {...field}
                />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="name"
          render={({ field }) => (
            <FormItem>
              <FormLabel>
                {admin.users.inviteDialog.nameLabel}{" "}
                <span className="text-muted-foreground">{common.terms.optional}</span>
              </FormLabel>
              <FormControl>
                <Input
                  autoComplete="name"
                  data-testid="invite-user-name"
                  {...field}
                />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit" disabled={isPending} className="w-full">
          {isPending
            ? common.forms.submit.sending
            : admin.users.inviteDialog.submit}
        </Button>
      </form>
    </Form>
  );
}

"use client";

import { useTransition } from "react";
import { toast } from "sonner";

import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/alert-dialog";
import { Button } from "@/components/ui/button";
import { revokeUserAction } from "@/features/users/actions";
import { useMessages } from "@/stores/use-content-store";

type Props = {
  userId: string;
  email: string;
  disabled?: boolean;
};

export function RevokeUserButton({ userId, email, disabled }: Props) {
  const { admin } = useMessages();
  const [isPending, startTransition] = useTransition();

  function onConfirm() {
    startTransition(async () => {
      const result = await revokeUserAction({ userId });
      if (result.success) {
        toast.success(admin.users.revokeDialog.success);
      } else {
        toast.error(result.error);
      }
    });
  }

  return (
    <AlertDialog>
      <AlertDialogTrigger asChild>
        <Button
          variant="ghost"
          size="sm"
          disabled={disabled || isPending}
          data-testid={`revoke-user-${userId}`}
        >
          {admin.users.revokeDialog.confirm}
        </Button>
      </AlertDialogTrigger>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>{admin.users.revokeDialog.title}</AlertDialogTitle>
          <AlertDialogDescription>
            {admin.users.revokeDialog.description(email)}
          </AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter>
          <AlertDialogCancel>{admin.users.revokeDialog.cancel}</AlertDialogCancel>
          <AlertDialogAction onClick={onConfirm}>
            {admin.users.revokeDialog.confirm}
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  );
}
